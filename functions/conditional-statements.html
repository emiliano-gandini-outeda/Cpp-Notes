The switch statement is used to choose one block of code from many, based on a single variable.

üîπ Syntax

    case value1:
        // code block
        break;
    case value2:
        // code block
        break;
    ...

        // code if no cases match
}

‚úÖ Example

    int day;
    cout << "Enter a number (1‚Äì7): ";
    cin >> day;

    switch (day) {
        case 1: cout << "Monday" << endl; break;

        case 4: cout << "Thursday" << endl; break;

        case 6: cout << "Saturday" << endl; break;
        case 7: cout << "Sunday" << endl; break;
        default: cout << "Invalid day number." << endl;
    }

    return 0;
}
