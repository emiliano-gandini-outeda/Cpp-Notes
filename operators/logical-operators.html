Logical operators are used to combine or modify conditions. The result is always either true (1) or false (0).

âœ… Example Code

#include <stdio.h>

int main() {
    int age = 20;

    if (age >= 18 && age <= 25) {
        printf("You are a young adult.\n");
    }

    if (age < 0 || age > 120) {
        printf("Invalid age.\n");
    }

    if (!(age == 30)) {
        printf("You are not 30 years old.\n");
    }

    return 0;
}

âœ… Example

#include <stdio.h>

int main() {
    int number;
    printf("Enter a number: ");
    scanf("%d", &number);

    if (number > 0) {
        printf("Positive number\n");
    } else if (number == 0) {
        printf("Zero\n");
    } else {
        printf("Negative number\n");
    }

    return 0;
}
