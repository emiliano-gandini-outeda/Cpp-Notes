Operator Meaning Example Result

‚úÖ Example Code

#include <stdio.h>

int main() {
    int a = 5, b = 10;

    printf("a == b: %d\n", a == b);  // 0 (false)
    printf("a != b: %d\n", a != b);  // 1 (true)
    printf("a > b: %d\n", a > b);    // 0
    printf("a < b: %d\n", a < b);    // 1
    printf("a >= b: %d\n", a >= b);  // 0
    printf("a <= b: %d\n", a <= b);  // 1

    return 0;
}
